const getSuggestions = (input, activeMissions) => {
  const [cmd, ...args] = input.split(' ');
  const baseCommands = ['/prune', '/boost', '/mirror', '/isolate', '/respawn', '/panic', '/run'];
  
  if (cmd === '/run') return ['morning_sweep', 'expansion_pack', 'resource_conserve'];

// A simplified D3 hook for the Liquidity River
import * as d3 from 'd3';
import { sankey, sankeyLinkHorizontal } from 'd3-sankey';

export const drawSankey = (svgRef, data, dimensions) => {
  const { width, height } = dimensions;
  const svg = d3.select(svgRef);
  svg.selectAll("*").remove();

  const generator = sankey()
    .nodeWidth(15)
    .nodePadding(10)
    .extent([[1, 1], [width - 1, height - 5]]);

  const graph = generator(data);

  // Draw Links (The flowing money)
  svg.append("g")
    .attr("fill", "none")
    .attr("stroke", "#10b981") // Emerald-500
    .attr("stroke-opacity", 0.4)
    .selectAll("path")
    .data(graph.links)
    .join("path")
      .attr("d", sankeyLinkHorizontal())
      .attr("stroke-width", d => Math.max(1, d.width));

  // Draw Nodes (The Accounts/Sinks)
  svg.append("g")
    .selectAll("rect")
    .data(graph.nodes)
    .join("rect")
      .attr("x", d => d.x0)
      .attr("y", d => d.y0)
      .attr("height", d => d.y1 - d.y0)
      .attr("width", d => d.x1 - d.x0)
      .attr("fill", "#064e3b");
};

Here is the professional breakdown of why R3F wins for your use case:

1. State Management (Zustand/React Integration)
R3F: It is "React-native" to the core. If a value in your Zustand store changes (e.g., a threat level in The Sentry), the R3F component re-renders or updates its properties instantly without any "bridge" logic.

A-Frame: It uses an Entity-Component-System (ECS) that exists outside the React lifecycle. Syncing React state to A-Frame requires "wrappers" that often become clunky when you have 18 departments worth of data flowing in.

R3F: It allows for instanced rendering. If you want to render 5,000 "Alpha Particles" in the Hunter's Nebula, R3F handles this with minimal overhead.

A-Frame: It creates a DOM element for every entity. 5,000 entities would likely tank the frame rate of your entire "OS" interface, especially when other React widgets are running simultaneously.

3. Layout: The "Legacy OS" Constraint
R3F: It fits perfectly inside a <div>. You can have multiple <Canvas> elements on one page (one for the Sentry, one for the Physicist) as resizable, draggable widgets.

A-Frame: It traditionally wants to take over the whole screen or use an <iframe>. Managing multiple A-Frame scenes as small, draggable Photoshop-style windows is a technical headache.
